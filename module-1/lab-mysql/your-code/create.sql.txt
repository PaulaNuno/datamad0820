CREATE TABLE `new_cars_company`.`cars` (
  `idcars` INT NOT NULL,
  `VIN` VARCHAR(45) NOT NULL,
  `manufacturer` VARCHAR(45) NOT NULL,
  `model` VARCHAR(45) NOT NULL,
  `year` YEAR NOT NULL,
  `color` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`idcars`));


CREATE TABLE `new_cars_company`.`customers` (
  `customers` INT NOT NULL,
  `customer_id` VARCHAR(45) NOT NULL,
  `phone_number` INT NOT NULL,
  `email` VARCHAR(45) NOT NULL,
  `address` VARCHAR(45) NOT NULL,
  `city` VARCHAR(45) NOT NULL,
  `state/province` VARCHAR(45) NOT NULL,
  `zip` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`customers`));

CREATE TABLE `new_cars_company`.`salespersons` (
  `staff_ID` INT NOT NULL,
  `name` VARCHAR(45) NOT NULL,
  `store_at_your_company` VARCHAR(45) CHARACTER SET 'ascii' NOT NULL,
  PRIMARY KEY (`staff_ID`));

CREATE TABLE `new_cars_company`.`Invoices` (
  `invoices` INT NOT NULL,
  `invoice_number` VARCHAR(45) NOT NULL,
  `date` DATE NOT NULL,
  `car` VARCHAR(45) NOT NULL,
  `customer` VARCHAR(45) NOT NULL,
  `salesperson_to_each_sale` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`invoices`));
